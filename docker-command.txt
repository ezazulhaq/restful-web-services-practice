# Steps to build and run a container
# Step 1 - Build image
docker build -t restfulwebservices:1.0 .
# Step 2 - Run Container
docker container run --name restservices -d -p 8081:8081 restfulwebservices:1.0
# Step 3 - Tag Image
docker tag restfulwebservices:1.0 ezazulhaq02/restfulwebservices:1.0
# Push tagged image to Docker hub 
docker push ezazulhaq02/restfulwebservices:1.0
